---
title: 'Create a new assistant'
api: 'POST /v0/assistant'
---
Creates a new assistant for the authenticated user with the specified configuration. This action will deduct credits from the user account.

#### Authorizations

<ParamField header="Authorization" type="string" required="true">
  Bearer authentication header of the form `Bearer <token>`, where `<token>` is your auth token.
</ParamField>

#### Body

<ParamField body="first_name" type="string | null" >
First name of the assistant
</ParamField>

<ParamField body="surname" type="string | null" >
Surname of the assistant
</ParamField>

<ParamField body="age" type="integer | null" >
Age of the assistant
</ParamField>

<ParamField body="weekly_limit" type="number | null" >
Weekly time limit for the assistant in hours
</ParamField>

<ParamField body="max_parallel" type="integer | null" >
Maximum number of parallel tasks the assistant can handle
</ParamField>

<ParamField body="region" type="string | null" >
Geographic region of the assistant
</ParamField>

<ParamField body="profile_photo" type="string | null" >
URL to the assistant's profile photo
</ParamField>

<ParamField body="about" type="string | null" >
Brief description about the assistant
</ParamField>

<ParamField body="email" type="string | null" >
Email of the assistant
</ParamField>

<ParamField body="voice_id" type="string | null" >
Id of the voice (Cartesia ID) to use for the assistant
</ParamField>

<ParamField body="user_phone" type="string | null" >
Contact phone number of the user
</ParamField>

<ParamField body="create_infra" type="boolean | null"  default={true}>
Whether to create the infrastructure for the assistant
</ParamField>

```bash cURL
    "voice_id": "bf0a246a-8642-498a-9950-80c35e9276b5",

}'
```

